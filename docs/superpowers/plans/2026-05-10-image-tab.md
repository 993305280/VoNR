> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为"素材库列表"的图片 tab 实现完整的后端 API 和前端对接，包括文件上传、CRUD 操作、分页查询和批量删除功能。

**Architecture:** 后端采用 MVC 架构（Model-Service-Controller），使用 multer 处理文件上传，sharp 提取图片分辨率。前端抽取为独立组件 ImageTab.vue + useImageData.js composable，与其他 tab 保持一致架构。

**Tech Stack:** Node.js, Express, MySQL (mysql2), multer, sharp, Vue 3, Pinia, Element Plus

const pool = require('../config/db');

const ImageModel = {
  // 分页查询图片列表
  async findWithPagination({ page = 1, pageSize = 10, name, auditStatus, available }) {
    let whereConditions = [];
    let params = [];

    if (name) {
      whereConditions.push('name LIKE ?');
      params.push(`%${name}%`);
    }
    if (auditStatus) {
      whereConditions.push('audit_status = ?');
      params.push(auditStatus);
    }
    if (available !== undefined && available !== '') {
      whereConditions.push('available = ?');
      params.push(Number(available));
    }

    const whereClause = whereConditions.length > 0 ? 'WHERE ' + whereConditions.join(' AND ') : '';

    // 查询总数
    const [countResult] = await pool.query(
      `SELECT COUNT(*) as total FROM images ${whereClause}`,
      params
    );
    const total = countResult[0].total;

    // 查询分页数据
    const offset = (page - 1) * pageSize;
    const [rows] = await pool.query(
      `SELECT * FROM images ${whereClause} ORDER BY created_at DESC LIMIT ? OFFSET ?`,
      [...params, pageSize, offset]
    );

    return { list: rows, total, page, pageSize };
  },

  // 根据 ID 查询
  async findById(id) {
    const [rows] = await pool.query('SELECT * FROM images WHERE id = ?', [id]);
    return rows[0] || null;
  },

  // 新增图片
  async create(data) {
    const [result] = await pool.query(
      `INSERT INTO images (name, file_path, file_name, resolution, format, file_size, audit_type, audit_status, sync_status, available, description, operator)
       VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
      [data.name, data.filePath, data.fileName, data.resolution, data.format, data.fileSize, data.auditType || '新增', data.auditStatus || '审核中', data.syncStatus || '未同步', data.available !== undefined ? data.available : 1, data.description || '', data.operator || '']
    );
    return { id: result.insertId, ...data };
  },

  // 更新图片
  async update(id, data) {
    const fields = [];
    const params = [];

    if (data.name !== undefined) { fields.push('name = ?'); params.push(data.name); }
    if (data.filePath !== undefined) { fields.push('file_path = ?'); params.push(data.filePath); }
    if (data.fileName !== undefined) { fields.push('file_name = ?'); params.push(data.fileName); }
    if (data.resolution !== undefined) { fields.push('resolution = ?'); params.push(data.resolution); }
    if (data.format !== undefined) { fields.push('format = ?'); params.push(data.format); }
    if (data.fileSize !== undefined) { fields.push('file_size = ?'); params.push(data.fileSize); }
    if (data.description !== undefined) { fields.push('description = ?'); params.push(data.description); }
    if (data.available !== undefined) { fields.push('available = ?'); params.push(data.available); }

    if (fields.length === 0) return null;

    params.push(id);
    await pool.query(`UPDATE images SET ${fields.join(', ')} WHERE id = ?`, params);
    return await this.findById(id);
  },

  // 删除图片
  async delete(id) {
    const image = await this.findById(id);
    if (!image) return null;
    await pool.query('DELETE FROM images WHERE id = ?', [id]);
    return image;
  },

  // 批量删除
  async batchDelete(ids) {
    if (!ids.length) return 0;
    const [result] = await pool.query('DELETE FROM images WHERE id IN (?)', [ids]);
    return result.affectedRows;
  }
};

module.exports = ImageModel;

```javascript
const fs = require('fs');
const path = require('path');
const sharp = require('sharp');
const ImageModel = require('../models/imageModel');

const UPLOAD_DIR = path.join(__dirname, '..', 'uploads', 'images');

const ImageService = {
  // 获取图片列表
  async getImages(params) {
    const result = await ImageModel.findWithPagination(params);
    // 转换文件路径为完整 URL
    result.list = result.list.map(item => this.formatImageItem(item));
    return result;
  },

  // 获取单个图片详情
  async getImageById(id) {
    const item = await ImageModel.findById(id);
    if (!item) return null;
    return this.formatImageItem(item);
  },

  // 新增图片
  async createImage(file, data) {
    // 提取图片分辨率
    let resolution = '';
    try {
      const metadata = await sharp(file.path).metadata();
      resolution = `${metadata.width}×${metadata.height}`;
    } catch (err) {
      console.error('提取分辨率失败:', err);
    }

    // 构建存储路径
    const now = new Date();
    const dateDir = path.join(
      String(now.getFullYear()),
      String(now.getMonth() + 1).padStart(2, '0'),
      String(now.getDate()).padStart(2, '0')
    );
    const ext = path.extname(file.originalname);
    const fileName = `${Date.now()}-${Math.random().toString(36).substr(2, 9)}${ext}`;
    const filePath = path.join('images', dateDir, fileName);

    // 确保目录存在
    const fullDir = path.join(UPLOAD_DIR, dateDir);
    if (!fs.existsSync(fullDir)) {
      fs.mkdirSync(fullDir, { recursive: true });
    }

    // 移动文件到目标位置
    const fullPath = path.join(UPLOAD_DIR, dateDir, fileName);
    fs.renameSync(file.path, fullPath);

    // 获取文件格式
    const format = ext.replace('.', '').toLowerCase();

    const imageData = {
      name: data.name,
      filePath: `/uploads/${filePath}`,
      fileName: file.originalname,
      resolution,
      format,
      fileSize: file.size,
      auditType: '新增',
      auditStatus: '审核中',
      syncStatus: '未同步',
      available: 1,
      description: data.description || '',
      operator: data.operator || ''
    };

    const result = await ImageModel.create(imageData);
    return this.formatImageItem(result);
  },

  // 更新图片
  async updateImage(id, data, file) {
    const existing = await ImageModel.findById(id);
    if (!existing) return null;

    const updateData = { name: data.name, description: data.description };

    // 如果上传了新文件
    if (file) {
      // 删除旧文件
      const oldPath = path.join(__dirname, '..', existing.file_path);
      if (fs.existsSync(oldPath)) {
        fs.unlinkSync(oldPath);
      }

      // 提取分辨率
      let resolution = '';
      try {
        const metadata = await sharp(file.path).metadata();
        resolution = `${metadata.width}×${metadata.height}`;
      } catch (err) {
        console.error('提取分辨率失败:', err);
      }

      // 构建新存储路径
      const now = new Date();
      const dateDir = path.join(
        String(now.getFullYear()),
        String(now.getMonth() + 1).padStart(2, '0'),
        String(now.getDate()).padStart(2, '0')
      );
      const ext = path.extname(file.originalname);
      const fileName = `${Date.now()}-${Math.random().toString(36).substr(2, 9)}${ext}`;
      const filePath = path.join('images', dateDir, fileName);

      // 确保目录存在
      const fullDir = path.join(UPLOAD_DIR, dateDir);
      if (!fs.existsSync(fullDir)) {
        fs.mkdirSync(fullDir, { recursive: true });
      }

      // 移动文件
      const fullPath = path.join(UPLOAD_DIR, dateDir, fileName);
      fs.renameSync(file.path, fullPath);

      updateData.filePath = `/uploads/${filePath}`;
      updateData.fileName = file.originalname;
      updateData.resolution = resolution;
      updateData.format = ext.replace('.', '').toLowerCase();
      updateData.fileSize = file.size;
      updateData.auditType = '编辑';
    }

    const result = await ImageModel.update(id, updateData);
    return this.formatImageItem(result);
  },

  // 删除图片
  async deleteImage(id) {
    const item = await ImageModel.findById(id);
    if (!item) return null;

    // 删除文件
    const filePath = path.join(__dirname, '..', item.file_path);
    if (fs.existsSync(filePath)) {
      fs.unlinkSync(filePath);
    }

    await ImageModel.delete(id);
    return this.formatImageItem(item);
  },

  // 批量删除
  async batchDeleteImages(ids) {
    // 获取所有要删除的图片信息
    const images = [];
    for (const id of ids) {
      const item = await ImageModel.findById(id);
      if (item) images.push(item);
    }

    // 删除文件
    for (const item of images) {
      const filePath = path.join(__dirname, '..', item.file_path);
      if (fs.existsSync(filePath)) {
        fs.unlinkSync(filePath);
      }
    }

    return await ImageModel.batchDelete(ids);
  },

  // 格式化图片项（转换文件路径）
  formatImageItem(item) {
    if (!item) return null;
    return {
      id: item.id,
      name: item.name,
      filePath: item.file_path,
      fileName: item.file_name,
      resolution: item.resolution,
      format: item.format,
      fileSize: item.file_size,
      fileSizeFormatted: this.formatFileSize(item.file_size),
      auditType: item.audit_type,
      auditStatus: item.audit_status,
      syncStatus: item.sync_status,
      available: item.available === 1,
      description: item.description,
      operator: item.operator,
      createdAt: item.created_at,
      updatedAt: item.updated_at
    };
  },

  // 格式化文件大小
  formatFileSize(bytes) {
    if (bytes === 0) return '0 B';
    const k = 1024;
    const sizes = ['B', 'KB', 'MB', 'GB'];
    const i = Math.floor(Math.log(bytes) / Math.log(k));
    return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
  }
};

module.exports = ImageService;
```

```javascript
const ImageService = require('../services/imageService');

const ImageController = {
  // GET /api/v1/images
  async getList(req, res) {
    try {
      const { page, pageSize, name, auditStatus, available } = req.query;
      const result = await ImageService.getImages({
        page: parseInt(page) || 1,
        pageSize: parseInt(pageSize) || 10,
        name,
        auditStatus,
        available
      });

      return res.json({
        code: 200,
        message: '获取成功',
        data: result
      });
    } catch (error) {
      console.error('获取图片列表失败:', error);
      return res.status(500).json({
        code: 500,
        message: '服务器内部错误',
        data: null
      });
    }
  },

  // GET /api/v1/images/:id
  async getById(req, res) {
    try {
      const { id } = req.params;
      const item = await ImageService.getImageById(parseInt(id));

      if (!item) {
        return res.status(404).json({
          code: 404,
          message: '图片不存在',
          data: null
        });
      }

      return res.json({
        code: 200,
        message: '获取成功',
        data: item
      });
    } catch (error) {
      console.error('获取图片详情失败:', error);
      return res.status(500).json({
        code: 500,
        message: '服务器内部错误',
        data: null
      });
    }
  },

  // POST /api/v1/images
  async create(req, res) {
    try {
      if (!req.file) {
        return res.status(400).json({
          code: 400,
          message: '请上传图片文件',
          data: null
        });
      }

      const { name, description } = req.body;
      if (!name) {
        return res.status(400).json({
          code: 400,
          message: '请输入素材名称',
          data: null
        });
      }

      const operator = req.user?.username || 'admin@VoNR';
      const result = await ImageService.createImage(req.file, {
        name,
        description,
        operator
      });

      return res.json({
        code: 200,
        message: '上传成功',
        data: result
      });
    } catch (error) {
      console.error('创建图片失败:', error);
      return res.status(500).json({
        code: 500,
        message: '服务器内部错误',
        data: null
      });
    }
  },

  // PUT /api/v1/images/:id
  async update(req, res) {
    try {
      const { id } = req.params;
      const { name, description } = req.body;

      if (!name) {
        return res.status(400).json({
          code: 400,
          message: '请输入素材名称',
          data: null
        });
      }

      const result = await ImageService.updateImage(parseInt(id), {
        name,
        description
      }, req.file);

      if (!result) {
        return res.status(404).json({
          code: 404,
          message: '图片不存在',
          data: null
        });
      }

      return res.json({
        code: 200,
        message: '更新成功',
        data: result
      });
    } catch (error) {
      console.error('更新图片失败:', error);
      return res.status(500).json({
        code: 500,
        message: '服务器内部错误',
        data: null
      });
    }
  },

  // DELETE /api/v1/images/:id
  async delete(req, res) {
    try {
      const { id } = req.params;
      const result = await ImageService.deleteImage(parseInt(id));

      if (!result) {
        return res.status(404).json({
          code: 404,
          message: '图片不存在',
          data: null
        });
      }

      return res.json({
        code: 200,
        message: '删除成功',
        data: null
      });
    } catch (error) {
      console.error('删除图片失败:', error);
      return res.status(500).json({
        code: 500,
        message: '服务器内部错误',
        data: null
      });
    }
  },

  // POST /api/v1/images/batch-delete
  async batchDelete(req, res) {
    try {
      const { ids } = req.body;
      if (!ids || !Array.isArray(ids) || ids.length === 0) {
        return res.status(400).json({
          code: 400,
          message: '请选择要删除的图片',
          data: null
        });
      }

      const count = await ImageService.batchDeleteImages(ids);

      return res.json({
        code: 200,
        message: `成功删除 ${count} 张图片`,
        data: null
      });
    } catch (error) {
      console.error('批量删除失败:', error);
      return res.status(500).json({
        code: 500,
        message: '服务器内部错误',
        data: null
      });
    }
  }
};

module.exports = ImageController;

const express = require('express');
const router = express.Router();
const multer = require('multer');
const path = require('path');
const ImageController = require('../controllers/imageController');
const authMiddleware = require('../middleware/auth');

// 配置 multer
const storage = multer.diskStorage({
  destination: (req, file, cb) => {
    cb(null, path.join(__dirname, '..', 'uploads', 'images'));
  },
  filename: (req, file, cb) => {
    const ext = path.extname(file.originalname);
    cb(null, `${Date.now()}-${Math.random().toString(36).substr(2, 9)}${ext}`);
  }
});

const fileFilter = (req, file, cb) => {
  const allowedTypes = ['image/png', 'image/jpeg', 'image/jpg'];
  if (allowedTypes.includes(file.mimetype)) {
    cb(null, true);
  } else {
    cb(new Error('只能上传 png/jpg 格式的图片'), false);
  }
};

const upload = multer({
  storage,
  fileFilter,
  limits: { fileSize: 10 * 1024 * 1024 } // 10MB
});

// 获取图片列表
router.get('/images', authMiddleware, ImageController.getList);

// 获取图片详情
router.get('/images/:id', authMiddleware, ImageController.getById);

// 新增图片
router.post('/images', authMiddleware, upload.single('file'), ImageController.create);

// 更新图片
router.put('/images/:id', authMiddleware, upload.single('file'), ImageController.update);

// 删除图片
router.delete('/images/:id', authMiddleware, ImageController.delete);

// 批量删除
router.post('/images/batch-delete', authMiddleware, ImageController.batchDelete);

module.exports = router;
```

## Task 8: 前端 Composable

**Files:**
- Create: `src/composables/useImageData.js`

- [ ] **Step 1: 创建 useImageData.js**

```javascript
import { ref, reactive } from 'vue'
import { getImageList, batchDeleteImages } from '@/api/image'
import { ElMessage } from 'element-plus'

export function useImageData() {
  const tableData = ref([])
  const loading = ref(false)
  const total = ref(0)
  const currentPage = ref(1)
  const pageSize = ref(10)
  const selectedRows = ref([])

  const searchForm = reactive({
    name: '',
    auditStatus: '',
    useStatus: ''
  })

  const fetchTableData = async () => {
    loading.value = true
    try {
      const params = {
        page: currentPage.value,
        pageSize: pageSize.value
      }
      if (searchForm.name) params.name = searchForm.name
      if (searchForm.auditStatus) params.auditStatus = searchForm.auditStatus
      if (searchForm.useStatus !== '') params.available = searchForm.useStatus === '可用' ? 1 : 0

      const res = await getImageList(params)
      tableData.value = res.data.list
      total.value = res.data.total
    } catch (error) {
      console.error('获取图片列表失败:', error)
      ElMessage.error('获取图片列表失败')
    } finally {
      loading.value = false
    }
  }

  const handleSearch = () => {
    currentPage.value = 1
    fetchTableData()
  }

  const handleReset = () => {
    searchForm.name = ''
    searchForm.auditStatus = ''
    searchForm.useStatus = ''
    handleSearch()
  }

  const handlePageChange = (page) => {
    currentPage.value = page
    fetchTableData()
  }

  const handleSizeChange = (size) => {
    pageSize.value = size
    currentPage.value = 1
    fetchTableData()
  }

  const handleSelectionChange = (rows) => {
    selectedRows.value = rows
  }

  const handleBatchDelete = async () => {
    if (selectedRows.value.length === 0) {
      ElMessage.warning('请先选择要删除的图片')
      return false
    }

    try {
      const ids = selectedRows.value.map(row => row.id)
      await batchDeleteImages(ids)
      ElMessage.success(`成功删除 ${ids.length} 张图片`)
      selectedRows.value = []
      fetchTableData()
      return true
    } catch (error) {
      console.error('批量删除失败:', error)
      ElMessage.error('批量删除失败')
      return false
    }
  }

  // 初始加载
  fetchTableData()

  return {
    tableData,
    loading,
    total,
    currentPage,
    pageSize,
    selectedRows,
    searchForm,
    fetchTableData,
    handleSearch,
    handleReset,
    handlePageChange,
    handleSizeChange,
    handleSelectionChange,
    handleBatchDelete
  }
}
```
